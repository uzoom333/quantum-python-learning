# requirements.txt
# Ambiente de Computação Quântica com Qiskit
# Instale com: pip install -r requirements.txt

# === CORE QISKIT ===
qiskit                    # SDK principal
qiskit-ibm-runtime        # Conexão com computadores quânticos reais da IBM
qiskit-aer                # Simulador local (roda no seu PC)

qiskit-machine-learning    # ML quântico

# === PYTHON CIENTÍFICO ===
numpy                      # Arrays e álgebra linear (base de TUDO)
scipy                      # Computação científica avançada
matplotlib                 # Gráficos e visualização
pandas                     # Manipulação de dados
sympy                      # Matemática simbólica

# === JUPYTER ===
jupyterlab                 # Interface de notebooks
ipykernel                  # Kernel Python pro Jupyter
ipywidgets                 # Widgets interativos

# === VISUALIZAÇÃO DE CIRCUITOS ===
pylatexenc                 # Renderiza circuitos quânticos bonitos
pillow                     # Processamento de imagens

# === DEV TOOLS ===
pytest                     # Testes
black                      # Formatação de código
flake8                     # Linting
